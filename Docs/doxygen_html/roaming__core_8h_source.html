65  if (state.E_translation>=r_parent->delta_E_roam) { // dissociation will occur
66  r_output_data->r_data->num_radical+=state.degeneracy;

68

70  state_data.E_trans=state.E_translation-r_parent->delta_E_roam; //at the potential of dissociation products (deltaEroam higher in energ) there will be deltaEroam less translational energy (E_avail is deltaEroam less for a traditional PST count)
71  state_data.vel_f1=r_frag1->get_velocity(state_data.E_trans);
72  state_data.vel_f2=r_frag2->get_velocity(state_data.E_trans);

75  state_data.E_vib_f1=r_frag1->vib_levels[state.f1.v].energy;
76  state_data.E_vib_f2=r_frag2->vib_levels[state.f2.v].energy;
77  state_data.E_int_f1=state_data.E_vib_f1+state.f1.E_nk+r_frag1->spin_orbit_states[state.f1.so].energy; //E_total=E_vib+E_rot+E_soc
78  state_data.E_int_f2=state_data.E_vib_f2+state.f2.E_nk+r_frag2->spin_orbit_states[state.f2.so].energy; //E_total=E_vib+E_rot+E_soc
79 
80  increment_histograms(r_output_data->r_data, state, state_data);
81  for (p=0; p<probe_ids.size(); p++) increment_histograms(r_output_data->r_probes[probe_ids[p]]->r_data, state, state_data);
82  }
83 
84  }else{ // may roam

88  if (!r_frag1->single_atom){//doubt we need the if here as the histgram won't increment if it isn't initialised, which the user shouldn't want and the code shouldn't allow ... revisit this! efficiency here counts!
89  r_output_data->r_data->frag1_nk.increment_value(state.f1.n, state.f1.k, state.degeneracy);
