91 bool Phasespace_Core::pass_centrifugal_barrier(const unsigned L, const double E_translation){
92  bool result=true;
93  if (parent->b_max>0) {
94  result= (parent->b_max*METERS_IN_ANGSTROM) >= ( (L*HBAR)/(reduced_mass*KG_IN_AMU*get_relative_velocity(E_translation)) ); // b >= L / (Mu*V)
95  }else if (parent->C_6>0) {
96  result= (L*(L+1)*HBARSD) <= (3*reduced_mass*KG_IN_AMU*pow( (2*parent->C_6*WAVENUMBER_ANGSTROM6_TO_JOULE_METER6*pow(E_translation*JOULES_IN_WAVENUMBER,2)) ,1.0/3.0)); // (L(L+1)hbar^2) <= ( 3Mu(3C_6*E_t^2)^(1/3) )
97  }
