74 double Geom_Molecule::get_angle(const unsigned &a, const unsigned &b, const unsigned &c) { //using the cosine rule, returns the angle abc defined by 3 atoms where ab and bc define the rays and b is at the vertex, in radians
75  double ab, bc, ac, theta;
76  ab=sqrt(pow((atoms[a].coord[0]-atoms[b].coord[0]),2)+pow((atoms[a].coord[1]-atoms[b].coord[1]),2)+pow((atoms[a].coord[2]-atoms[b].coord[2]),2));
77  bc=sqrt(pow((atoms[b].coord[0]-atoms[c].coord[0]),2)+pow((atoms[b].coord[1]-atoms[c].coord[1]),2)+pow((atoms[b].coord[2]-atoms[c].coord[2]),2));
78  ac=sqrt(pow((atoms[a].coord[0]-atoms[c].coord[0]),2)+pow((atoms[a].coord[1]-atoms[c].coord[1]),2)+pow((atoms[a].coord[2]-atoms[c].coord[2]),2));
79  theta=acos((ab*ab+bc*bc-ac*ac)/(2*ab*bc));
80  return theta;
81 }

279 void Geom_Molecule::get_inertia_tensor() {
280  /*---------------------------
281  The inertia tensor looks like this
282  | Ixx Ixy Ixz |
283  | Iyx Iyy Iyz |
284  | Izx Izy Izz |
285  ---------------------------*/
286  unsigned i,j;
287  TNT::Array2D <double> inertia_eigenvectors(3,3);
288  TNT::Array1D <double> inertia_eigenvalues(3);
289  double Ixx, Iyy, Izz, Ixy, Ixz, Iyz, eigencheck1, eigencheck2;
290  Geom_Eigen_Data temp_eigen_data;
291  Ixx=Iyy=Izz=Ixy=Ixz=Iyz=0;
292  for (i=0;i<atoms.size();i++){
293  Ixx+=atoms[i].mass*(pow(atoms[i].com_coord[1],2)+pow(atoms[i].com_coord[2],2)); //Ixx+=mass*(y^2+z^2)
294  Iyy+=atoms[i].mass*(pow(atoms[i].com_coord[0],2)+pow(atoms[i].com_coord[2],2)); //Iyy+=mass*(x^2+z^2)
295  Izz+=atoms[i].mass*(pow(atoms[i].com_coord[0],2)+pow(atoms[i].com_coord[1],2)); //Izz+=mass*(x^2+y^2)
296  Ixy-=atoms[i].mass*atoms[i].com_coord[0]*atoms[i].com_coord[1]; //Ixy-=mass*x*y
297  Ixz-=atoms[i].mass*atoms[i].com_coord[0]*atoms[i].com_coord[2]; //Ixz-=mass*x*z
298  Iyz-=atoms[i].mass*atoms[i].com_coord[1]*atoms[i].com_coord[2]; //Iyz-=mass*y*z
299  }
300  inertia_tensor[0][0]=Ixx;
301  inertia_tensor[1][1]=Iyy;
302  inertia_tensor[2][2]=Izz;
303  inertia_tensor[0][1]=inertia_tensor[1][0]=Ixy;
304  inertia_tensor[0][2]=inertia_tensor[2][0]=Ixz;
305  inertia_tensor[1][2]=inertia_tensor[2][1]=Iyz;
306 
307  JAMA::Eigenvalue <double> eigens(inertia_tensor);
308  eigens.getV(inertia_eigenvectors); //results are slightly different from old code, MJTJ said that a factor of -1 difference is common as it is algorith dependent, so it will be consistent, and we may need to modify the results to match ...
309  eigens.getRealEigenvalues(inertia_eigenvalues);
310 
311  for (i=0;i<3;i++){
312  inertia_eigendata[i].val=abs(inertia_eigenvalues[i]);
313  for (j=0;j<3;j++) inertia_eigendata[i].vect[j]=inertia_eigenvectors[j][i];
314  }
315  //JAMA::Eigenvalue returns the results in ascending order, which is what we need for the (eigencheck2<eigencheck1) swap to work
316  eigencheck1=inertia_eigendata[1].val/inertia_eigendata[0].val;
317  eigencheck2=inertia_eigendata[2].val/inertia_eigendata[1].val;
318  //************************* SORT SO THAT THE LEAST SIMILAR ONE IS ISOLATED in [2]! *************************
319  if (eigencheck2<eigencheck1){
320  temp_eigen_data=inertia_eigendata[2];
321  inertia_eigendata[2]=inertia_eigendata[0];
322  inertia_eigendata[0]=temp_eigen_data;
323  }
324  arithmetic_mean_I_tensor=(inertia_eigendata[0].val+inertia_eigendata[1].val+inertia_eigendata[2].val)/3; // (IA+IB+IC)/3
325  geometric_mean_I_tensor=cbrt(inertia_eigendata[0].val*inertia_eigendata[1].val*inertia_eigendata[2].val); // (IA*IB*IC)^(1/3)
326  bcmplx_tensor=cbrt( (16.86/inertia_eigendata[0].val) * (16.86/inertia_eigendata[1].val) * (16.86/inertia_eigendata[2].val) ); //(16.86/IA * 16.86/IB * 16.86 /IC)^(1/3)
327  if (inertia_eigendata[2].val <= 0.05) no_torsion_tensor=1;
328  if (inertia_eigendata[0].val <= 0.05 || inertia_eigendata[1].val <= 0.05) no_2d_rotor_tensor=1;
329  //I_AMU_SQANGSTROM_TO_B_WAVENUMBER is a conversion factor between AMU.A^2 to CM^-1 ... turns I into B
330  geometric_B_tensor=I_AMU_SQANGSTROM_TO_B_WAVENUMBER/sqrt(inertia_eigendata[0].val*inertia_eigendata[1].val);
331 }

348  for (i=0;i<target_atoms.size();i++){

353  Ixz-=atoms[target_atoms[i]].mass*(atoms[target_atoms[i]].coord[0]-centre_of_mass[0])*(atoms[target_atoms[i]].coord[2]-centre_of_mass[2]); //Ixz-=mass*x*z
354  Iyz-=atoms[target_atoms[i]].mass*(atoms[target_atoms[i]].coord[1]-centre_of_mass[1])*(atoms[target_atoms[i]].coord[2]-centre_of_mass[2]); //Iyz-=mass*y*z
355  }
356  sub_inertia_tensor[0][0]=Ixx;
357  sub_inertia_tensor[1][1]=Iyy;
358  sub_inertia_tensor[2][2]=Izz;
359  sub_inertia_tensor[0][1]=sub_inertia_tensor[1][0]=Ixy;
360  sub_inertia_tensor[0][2]=sub_inertia_tensor[2][0]=Ixz;
361  sub_inertia_tensor[1][2]=sub_inertia_tensor[2][1]=Iyz;
362  JAMA::Eigenvalue <double> eigens(sub_inertia_tensor);
363  eigens.getV(inertia_eigenvectors); //results are slightly different from old code, MJTJ said that a factor of -1 difference is common as it is algorith dependent, so it will be consistent, and we may need to modify the results to match ...
364  eigens.getRealEigenvalues(inertia_eigenvalues);
365  for (i=0;i<3;i++){
366  inertia_results[i].val=abs(inertia_eigenvalues[i]);
367  for (j=0;j<3;j++) inertia_results[i].vect[j]=inertia_eigenvectors[j][i];
368  }
369  //JAMA::Eigenvalue returns the results in ascending order, which is what we need for the (eigencheck2<eigencheck1) swap to work
370  eigencheck1=inertia_results[1].val/inertia_results[0].val;
371  eigencheck2=inertia_results[2].val/inertia_results[1].val;
372  //************************* SORT SO THAT THE LEAST SIMILAR ONE IS ISOLATED in [2]! *************************
373  if (eigencheck2<eigencheck1){
374  temp_eigen_data=inertia_results[2];
375  inertia_results[2]=inertia_results[0];
376  inertia_results[0]=temp_eigen_data;
377  }
378  return inertia_results;
379 }
