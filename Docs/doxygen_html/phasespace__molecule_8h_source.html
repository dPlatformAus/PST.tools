110 void Combination_Band_List::enumerate(const double E_available, std::vector<unsigned> &combo_band, unsigned length_remaining, unsigned in_i) {
111  unsigned i, j;
112  Vibrational_Level combo_result;
113  length_remaining-=1; //length remaining is one less after adding this character
114  for (i=in_i; i < modes.size(); i++) { //try adding all the characters in the selection
115  combo_band.push_back(i); // add this energy level to the combination (*** store the index ***)
116  combo_result.degeneracy=1;
117  combo_result.energy=0;
118  for (j=0; j<combo_band.size(); j++){ //for each energy level in this combination, calculate its contribution to the combination band
119  combo_result.degeneracy*=modes[ combo_band[j] ].degeneracy;
120  combo_result.energy+=modes[ combo_band[j] ].energy;
121  }
122  // can check this combination is within E_available - kill recursion when E_available is exceeded preventing too many unnessesary calculations
123  if (combo_result.energy<=E_available) { //this combo still fits into energy
124  make_name(combo_band, combo_result);
125  bands.push_back(combo_result); // store it
126  if (length_remaining) {//length_remaining > 0 so do next length too - kill recursion once max length met
127  enumerate(E_available, combo_band, length_remaining, i); //enumerate all starting with this combo, start from this normal mode so we get all combos starting with it repeating
128  }
129  }
130  combo_band.pop_back(); // remove this energy level from the combination
131  }//move to next energy level
132 }

400 unsigned Phasespace_Molecule::get_n_max(const double E_available) const { //caution: don't call this unless set_Bbar_and_prolate() has been called
401  // Calculate max N, occurs when: K=0 if prolate (like HCO), K=N if oblate (like CH3), 0 if single atom
402  if (single_atom) return 0;
403  else { // ceil is important here, don't want to exclude anything ... if the resultant nk state is too big it is rejected anyway, but if we don't test and it fits - we are wrong
404  if (prolate || linear) return (unsigned) ceil( (-Bbar+sqrt(pow(Bbar,2)+4*Bbar*E_available))/(2*Bbar) ); //Max is BN(N+1) ie K=0 ... if linear==1 then prolate==1, so || is redundant, but it reads better
405  else return (unsigned) ceil( (-Bbar+sqrt(pow(Bbar,2)+4*A*E_available))/(2*A) ); //Max is BN(N+1)+(A-B)K^2 where N=K
406  }
407 }
